\section{Fase 2 — Bug Fixes Kritikal (17--25 Mar)}

\subsection{\fixbadge{FIX-1} Double Position Guard}
\textbf{Commit:} \commit{492cf0b} \hfill \textbf{Tanggal:} $\sim$17 Mar

\textbf{Masalah:} Bot dapat membuka posisi baru ketika posisi lama masih OPEN di exchange, mengakibatkan double exposure.

\textbf{Fix:} Cek \texttt{get\_open\_position()} sebelum setiap entry. Jika ada posisi aktif, skip entry.

\subsection{\fixbadge{FIX-2} Signal Replay setelah Restart}
\textbf{Commit:} \commit{d8e7a87} \hfill \textbf{Tanggal:} $\sim$17 Mar

\textbf{Masalah:} Setelah restart, \texttt{last\_notified\_ts} hilang dari memory sehingga sinyal lama dapat di-\textit{replay}.

\textbf{Fix:} Persist \texttt{last\_notified\_ts} ke disk (\texttt{last\_candle\_ts.json}).

\subsection{\fixbadge{FIX-3} SL Order Authentication}
\textbf{Commit:} \commit{c3107f9} \hfill \textbf{Tanggal:} $\sim$17 Mar

\textbf{Masalah:} SL order gagal ditempatkan karena auth header tidak dikirim pada endpoint order placement.

\textbf{Fix:} Semua order placement request menggunakan auth token yang valid.

\textbf{Impact:} Trade 18 Mar (-\$6.30) terjadi \textit{sebelum} fix ini — SL tidak terpasang sehingga posisi tidak terlindungi.

\subsection{\fixbadge{FIX-4} Advisory Signal Tidak Tereksekusi}
\textbf{Commit:} \commit{3160dc1} \hfill \textbf{Tanggal:} $\sim$18 Mar

\textbf{Masalah:} Signal dengan conviction 15--20\% (status ADVISORY) di-skip.

\textbf{Fix:} Update gate logic — ADVISORY signals diperlakukan sama dengan ACTIVE untuk entry.

\subsection{\fixbadge{FIX-5} Lighter Auth Token via SDK}
\textbf{Commit:} \commit{e28c312} \hfill \textbf{Tanggal:} 25 Mar

\textbf{Masalah:} \texttt{\_generate\_auth\_token()} membuat token format manual yang tidak valid — Lighter API membutuhkan token yang \textbf{ditandatangani} dengan private key.

\begin{lstlisting}[language=Python, caption=Auth Token Fix]
# Sebelum (BROKEN):
token = f"{expiry_unix}:{account_index}:{api_key_index}:{random_hex}"

# Sesudah (FIXED):
token_result = client.create_auth_token_with_expiry(
    deadline=lighter_sdk.SignerClient.DEFAULT_10_MIN_AUTH_EXPIRY,
    api_key_index=self.api_key_index,
)
token = token_result[0]  # SDK returns tuple
\end{lstlisting}

\subsection{\fixbadge{FIX-6} Exit Price dari trigger\_price}
\textbf{Commit:} \commit{e28c312} \hfill \textbf{Tanggal:} 25 Mar

\textbf{Masalah:} \texttt{fetch\_last\_closed\_order()} selalu return \texttt{filled\_price = 0.0} untuk SL/TP order karena Lighter menggunakan \textit{reduce-only trigger order} — \texttt{filled\_base\_amount = "0.00000"}.

\begin{lstlisting}[language=Python, caption=Exit Price Fix]
filled_base = float(last_order.get("filled_base_amount", 0) or 0)
if filled_base > 0 and filled_quote > 0:
    filled_price = filled_quote / filled_base  # actual fill
else:
    trigger_price = float(last_order.get("trigger_price", 0) or 0)
    filled_price = trigger_price if trigger_price > 0 else order_price
\end{lstlisting}

\subsection{\fixbadge{FIX-7} Startup Position Sync}
\textbf{Commit:} \commit{861822f} \hfill \textbf{Tanggal:} 25 Mar

\textbf{Masalah:} Restart saat posisi open menyebabkan state DB tidak sinkron dengan exchange.

\textbf{Fix:} Jalankan \texttt{sync\_position\_status()} saat daemon start.

\subsection{\fixbadge{FIX-8} Exit Type Detection dari order\_type}
\textbf{Commit:} \commit{861822f} \hfill \textbf{Tanggal:} 25 Mar

\textbf{Masalah:} Exit type (SL/TP) ditentukan dari \textit{jarak harga} — tidak reliable.

\begin{lstlisting}[language=Python, caption=Exit Type Detection Fix]
order_type_str = last_order.get("order_type", "").lower()
if "stop" in order_type_str:
    exit_type = "SL"
elif "take-profit" in order_type_str:
    exit_type = "TP"
\end{lstlisting}

\subsection{\fixbadge{FIX-9} L2 EMA Weakening Diperkuat}
\textbf{Commit:} \commit{8ab9b1b} \hfill \textbf{Tanggal:} 25 Mar

\textbf{Masalah:} L2 tidak cukup memblokir sinyal kontra-trend — confidence hanya dikurangi, tidak di-set \texttt{False}.

\textbf{Fix:} Force \texttt{l2 = False} saat L2 weakening aktif.

\subsection{\fixbadge{FIX-10} OCO Canceled Order Salah Dibaca sebagai SL}
\textbf{Commit:} \commit{f644601} \hfill \textbf{Tanggal:} 27 Mar

\textbf{Masalah:} Ketika TP hit, SL companion order di-cancel otomatis oleh exchange (OCO behavior). Order canceled ini memiliki \texttt{trigger\_price} valid sehingga bot mengira exit via SL.

\textbf{Akibat:}
\begin{itemize}[leftmargin=1.5em]
    \item Notifikasi Telegram menampilkan ``CLOSED — SL'' padahal TP hit
    \item \textbf{SL freeze aktif secara salah} $\rightarrow$ entry berikutnya diblokir
    \item Trade jam 03:00 WIB (27 Mar) tidak tereksekusi karena freeze palsu
\end{itemize}

\begin{lstlisting}[language=Python, caption=OCO Canceled Order Fix]
for last_order in closed_orders:
    status = last_order.get("status", "")
    # Skip canceled orders (OCO companion yang auto-canceled)
    if "cancel" in status.lower():
        continue
    # ... proses order yang benar-benar filled
\end{lstlisting}
